##### 建立Model和Controller

    ```php app\Models\News\News.php
    class News extends Model
    {
        use HasFactory;
        public $timestamps = false;
        protected $table = "news";
        protected $primaryKey = "id";
        protected $fillable = [
            "id",
            "typeId",   // 類別代碼
            "title",    // 標題
            "subtitle", // 次標題
            "dates",    // 日期
            "content",  // 內容
            "photo",    // 圖檔
            "createTime"    // 建立時間
        ];

    }
    ```

    ```php app\Http\Controllers\Admin\News\NewsTypeController.php
    
    // 前面use的路徑要注意，可能是不同的東西
    // 比如 Session 用錯就不能用flash() 功能了
    
    use App\Http\Controllers\Controller;
    use App\Models\News\NewsType;
    use Illuminate\Support\Facades\Session;
    use Illuminate\Http\Request;

    public function list()
    {
        $list = NewsType::get();
        return view("admin.news.type.list", compact("list"));
    }

    public function add()
    {
        return view("admin.news.type.add");
    }

    public function insert(Request $req)
    {
        $type = new NewsType();
        $type->title = $req->title;
        $type->save();

        Session::flash("message", "已新增");
        return redirect("admin/news/type/list");
    }
    ```

#### 最新消息頁面

##### 在Model裡增加新的getList()

    ```php app\Models\News\News.php
    public function getList(){

        // select a.*, b.title as types From news a INNER JOIN news_type b ON a.typeId = b.id limit 10

        $list = DB::table("news AS a")
        ->selectRaw("a.*, b.title as types")
        ->join("news_type AS b", "a.typeId","b.id")
        ->paginate(10);

        return $list;
    }
    ```
